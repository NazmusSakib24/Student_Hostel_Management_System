CREATE TABLE Users (
    UserID INT PRIMARY KEY IDENTITY,
    Username VARCHAR(50) NOT NULL,
    Password INT NOT NULL,
    Role VARCHAR(10) NOT NULL,  -- 'Admin', 'Staff', 'Student'
    SecurityAns VARCHAR(100) NOT NULL
);
 
CREATE TABLE Admins (
    AdminID INT PRIMARY KEY IDENTITY,
    UserID INT,
    Name VARCHAR(100),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

CREATE TABLE Rooms (
    RoomID INT PRIMARY KEY IDENTITY,
    RoomNumber VARCHAR(20),
    Capacity INT,
    Status VARCHAR(20)  -- 'Available', 'Occupied'
);
 
CREATE TABLE Students (
    StudentID INT PRIMARY KEY IDENTITY,
    UserID INT,
    Name VARCHAR(100),
    Phone VARCHAR(20),
    AssignedRoomID INT,
    FOREIGN KEY (UserID) REFERENCES Users(UserID),
    FOREIGN KEY (AssignedRoomID) REFERENCES Rooms(RoomID)
);
 
CREATE TABLE Staff (
    StaffID INT PRIMARY KEY IDENTITY,
    UserID INT,
    Name VARCHAR(100),
    Phone VARCHAR(20),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);
 
CREATE TABLE UtilityBills (
    BillID INT PRIMARY KEY IDENTITY,
    RoomID INT,
    Month VARCHAR(20),
    Electricity FLOAT,
    Water FLOAT,
    Gas FLOAT,
    Status VARCHAR(20),  -- 'Paid', 'Unpaid'
    FOREIGN KEY (RoomID) REFERENCES Rooms(RoomID)
);
 
CREATE TABLE ServiceRequests (
    RequestID INT PRIMARY KEY IDENTITY,
    StudentID INT,
    RoomID INT,
    Type VARCHAR(50),
    Description VARCHAR(MAX),
    Status VARCHAR(50) DEFAULT 'Pending',
    FOREIGN KEY (StudentID) REFERENCES Students(StudentID),
    FOREIGN KEY (RoomID) REFERENCES Rooms(RoomID)
);
